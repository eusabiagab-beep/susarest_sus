Esquema SQL listo para implementar en Supabase con Row Level Security (RLS)
-- ═══════════════════════════════════════
-- SUSAREST · Supabase Schema (PostgreSQL)
-- Row Level Security habilitado por tenant
-- ═══════════════════════════════════════

CREATE TABLE tenants (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  slug        TEXT UNIQUE NOT NULL,         -- 'steakhouse1'
  name        TEXT NOT NULL,
  logo_url    TEXT,
  plan        TEXT DEFAULT 'free',         -- free|pro|enterprise
  currency    TEXT DEFAULT 'MXN',
  tax_rate    DECIMAL(5,2) DEFAULT 16.00,
  fc_meta     DECIMAL(5,2) DEFAULT 28.00,
  fc_alert    DECIMAL(5,2) DEFAULT 32.00,
  fc_crit     DECIMAL(5,2) DEFAULT 35.00,
  margin_goal DECIMAL(5,2) DEFAULT 68.00,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE inventory_items (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id    UUID REFERENCES tenants(id) ON DELETE CASCADE,
  name         TEXT NOT NULL,
  category     TEXT,
  unit         TEXT NOT NULL,               -- kg, g, litro, ml, pieza
  stock        DECIMAL(12,4) DEFAULT 0,
  min_stock    DECIMAL(12,4) DEFAULT 0,
  cost         DECIMAL(12,4) NOT NULL,         -- por unidad base
  created_at   TIMESTAMPTZ DEFAULT NOW(),
  updated_at   TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_inv_tenant ON inventory_items(tenant_id);

CREATE TABLE price_history (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id   UUID REFERENCES tenants(id),
  item_id     UUID REFERENCES inventory_items(id),
  old_cost    DECIMAL(12,4),
  new_cost    DECIMAL(12,4),
  pct_change  DECIMAL(8,2),
  changed_at  TIMESTAMPTZ DEFAULT NOW(),
  changed_by  UUID
);
CREATE INDEX idx_ph_item ON price_history(item_id, changed_at);

CREATE TABLE recipes (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id   UUID REFERENCES tenants(id),
  name        TEXT NOT NULL,
  category    TEXT,
  portions    INT DEFAULT 1,
  sale_price  DECIMAL(12,2),
  is_active   BOOLEAN DEFAULT TRUE,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE recipe_ingredients (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id   UUID REFERENCES tenants(id),
  recipe_id   UUID REFERENCES recipes(id) ON DELETE CASCADE,
  item_id     UUID REFERENCES inventory_items(id),
  quantity    DECIMAL(12,4) NOT NULL,
  unit        TEXT,                              -- puede diferir del item.unit
  conv_factor DECIMAL(12,6) DEFAULT 1          -- conversión si aplica
);

CREATE TABLE food_cost_snapshots (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     UUID REFERENCES tenants(id),
  recipe_id     UUID REFERENCES recipes(id),
  total_cost    DECIMAL(12,4),
  cost_portion  DECIMAL(12,4),
  fc_pct        DECIMAL(8,4),
  margin_pct    DECIMAL(8,4),
  snapped_at    TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE purchase_orders (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id    UUID REFERENCES tenants(id),
  supplier     TEXT,
  status       TEXT DEFAULT 'pendiente',
  total        DECIMAL(12,2),
  delivery_at  DATE,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- ── ROW LEVEL SECURITY ──
ALTER TABLE inventory_items ENABLE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation ON inventory_items
  USING (tenant_id = current_setting('app.tenant_id')::UUID);

-- (Repetir para todas las tablas)
Supabase Edge Function (food cost automático):
Trigger en price_history → llama a recalculate_fc(item_id) → actualiza food_cost_snapshots → dispara notificación si FC supera meta.
